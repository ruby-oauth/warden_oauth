def build(keyword, config)
  strategy_class = create_oauth_strategy_class(keyword)
  register_oauth_strategy_class(keyword, strategy_class)
  set_oauth_service_info(strategy_class, config)
  # adding the access_token_user_finder to the strategy
  if access_token_user_finders.include?(keyword)
    strategy_class.access_token_user_finder(&access_token_user_finders[keyword])
  end
end
